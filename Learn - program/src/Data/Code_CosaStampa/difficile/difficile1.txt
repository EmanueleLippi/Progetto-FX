import java.util.Arrays;

public class Main {
    public static void main(String[] args) {
        int[] numbers = {5, 3, 8, 1, 2};
        Arrays.sort(numbers);

        int searchIndex = Arrays.binarySearch(numbers, 3);
        int sum = 0;

        for (int number : numbers) {
            sum += number;
        }

        System.out.println("searchIndex: " + searchIndex + ", sum: " + sum);
    }
}
