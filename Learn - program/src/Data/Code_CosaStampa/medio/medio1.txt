public class Main {
    public static void main(String[] args) {
        int[] array = {1, 2, 3, 4, 5};
        int result = 0;
        for (int i = 0; i < array.length; i++) {
            if (array[i] % 2 == 0) {
                result += array[i];
            } else {
                result -= array[i];
            }
        }
        System.out.println("result: " + result);
    }
}
